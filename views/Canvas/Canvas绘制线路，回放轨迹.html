Canvas绘制线路，回放轨迹

Q_ZHANG 2020-04-20 21:00:00 Canvas

应用场景：GIS，巡检，在线签名，作画等； 扩展：可以添加背景图片，融合业务


    // 实现原理 
    // 1：创建画布，获取鼠标相对位置，设置鼠标按下或松开的开关
    // 2：鼠标左键每按下一次就记录起点的位置，生成 o = { s: { x: 0, y: 0 }, point: [] } 数据格式，最后将多个位置拼成一个数组
    // 3：将数组post到服务端，并清空当前画布
    // 4：获取提交的数组，循环每一项的起点位置并moveTo，再进入到reDraw，递归point，实现定时回显每一笔，递归完成将进行下一次循环和递归，直到结束


<!DOCTYPE html>
    <html lang="en">

    <head>
        <meta charset="UTF-8">
        <meta name="viewport" content="width=device-width, initial-scale=1.0">
        <title>Document</title>
    </head>

    <body>
        <button onclick="startDrawLine()">作图</button>
        <button onclick="submit()">提交作图</button>
        <button onclick="draw(0,0)">回显</button>
        <canvas id="canvas" width="400" height="400" style="margin:1%;border: 1px solid blue;"></canvas>
        <script>
            var coordinates = [], can, ctx, isAllowDrawLine;
            function windowToCanvas(canvas, x, y) {
                //获取canvas元素距离窗口的一些属性
                let rect = canvas.getBoundingClientRect()
                return {
                    x: x - rect.left * (canvas.width / rect.width),
                    y: y - rect.top * (canvas.height / rect.height)
                }
            }
            //绘制路线
            function startDrawLine() {
                let self = this;
                can = document.getElementById("canvas");
                ctx = can.getContext("2d");
                //设置线条颜色
                ctx.strokeStyle = 'red';
                (function () {
                    if (!can || !can.getContext) {
                        return false
                    } else {
                        isAllowDrawLine = false;
                        //按下鼠标左键
                        can.onmousedown = function (e) {
                            var o = { s: { x: 0, y: 0 }, point: [] };
                            isAllowDrawLine = true;
                            coordinates.push(o);
                            let ele = self.windowToCanvas(can, e.clientX, e.clientY);
                            let { x, y } = ele;
                            (o.s.x = x, o.s.y = y);

                            ctx.moveTo(x, y);
                            console.log("获取点击值x", x);
                            console.log("获取点击值y", y);
                            // coordinates.push({ x: x, y: y })
                            ctx.beginPath();
                            can.onmousemove = (e) => {
                                if (isAllowDrawLine) {

                                    let ele = self.windowToCanvas(can, e.clientX, e.clientY);
                                    let { x, y } = ele;
                                    console.log("获取移动值x", x);
                                    console.log("获取移动值y", y);
                                    o.point.push({ x: x, y: y });

                                    ctx.lineTo(x, y)
                                    ctx.stroke()

                                }
                            }

                        };
                        console.log("获取位置=>>", self.pointsGather);
                        //松开鼠标时
                        can.onmouseup = function () {
                            isAllowDrawLine = false;
                        }
                    }
                })()
            }
            function submit() {
                (coordinates.length > 0) && (
                    console.log(coordinates),
                    can.onmousedown = null,
                    ctx.clearRect(0, 0, 400, 400)
                )

            }

            function draw(i) {
                ctx.lineWidth = 2;
                ctx.strokeStyle = "red";
                ctx.beginPath();
                if (i < coordinates.length) {
                    console.log(i)
                    var dTimer = setTimeout(() => {
                        ctx.moveTo(coordinates[i].s.x, coordinates[i].s.y);
                        reDraw(i, 0)
                        if (dTimer) {
                            clearTimeout(dTimer)
                            dTimer = null
                        }
                        ctx.closePath()
                    }, 500)
                } else {
                    return
                }

            }
            function reDraw(i, j) {
                if (j < coordinates[i].point.length) {
                    var rTimer = setTimeout(_ => {
                        ctx.lineTo(coordinates[i].point[j].x, coordinates[i].point[j].y)
                        ctx.stroke();
                        if (rTimer) {
                            clearTimeout(rTimer)
                            rTimer = null
                        }
                        j++
                        reDraw(i, j)
                    }, 30)
                } else {
                    i++
                    draw(i)
                    return false
                }
            }
        </script>
    </body>
